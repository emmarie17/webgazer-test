  <p>Look around the screen. WebGazer will try to estimate where you are looking.</p>

 

  <script>

    // Start WebGazer

    window.onload = function() {

      webgazer.setRegression('ridge') // choose prediction model

              .setGazeListener(function(data, elapsedTime) {

                if (data == null) return;

                var x = data.x; // x coordinate

                var y = data.y; // y coordinate

                console.log("X: " + x + " Y: " + y); // prints to browser console

              })

              .begin();

    };

  </script>
