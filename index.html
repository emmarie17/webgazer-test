  <p>Click each red dot a few times to calibrate. Then look around the screen and watch the green dot follow your gaze.</p>

 

  <div id="calibration"></div>

  <div id="tracker"></div>

 

  <script>

    // Generate calibration dots

    const calibration = document.getElementById("calibration");

    for (let i = 0; i < 9; i++) {

      const dot = document.createElement("div");

      dot.className = "dot";

      dot.addEventListener("click", () => {

        dot.style.background = "gray"; // mark as calibrated

      });

      calibration.appendChild(dot);

    }

 

    const tracker = document.getElementById("tracker");

 

    // Start WebGazer

    window.onload = function() {

      webgazer.setRegression('ridge')

              .setGazeListener(function(data, elapsedTime) {

                if (data == null) return;

                const x = data.x;

                const y = data.y;

                tracker.style.display = "block";

                tracker.style.left = (x - 10) + "px";

                tracker.style.top = (y - 10) + "px";

              })

              .begin()

              .showVideoPreview(true) // small camera view

              .showPredictionPoints(false); // don’t clutter

    };

  </script>
